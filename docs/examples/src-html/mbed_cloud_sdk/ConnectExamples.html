181    /**
182     * Creates, updates and deletes a presubscription.
183     */
184    @SuppressWarnings("null")
185    @Example
186    public void managePresubscription() {
187        ConnectionOptions config = Configuration.get();
188        Connect api = new Connect(config);
189        try {
190            // Listing already defined presubscriptions.
191            List<Presubscription> listPresubscriptions = api.listPresubscriptions();
192            for (Presubscription presubscription : listPresubscriptions) {
193                log("Already defined presubscription", presubscription);
194            }
195            // Getting a connected device.
196            DeviceListOptions options = new DeviceListOptions();
197            options.setLimit(Integer.valueOf(1));
198            Paginator<Device> deviceIterator = api.listAllConnectedDevices(options);
199            if (!deviceIterator.hasNext()) {
200                fail("No endpoints registered. Aborting.");
201            }
202            Device device = deviceIterator.next();
203            log("Device", device);
204            // Adding presubscriptions on all observable resources of a device.
205            List<Resource> observableResources = api.listObservableResources(device);
206            if (observableResources == null) {
207                fail("There is no observable resources on this device");
208            }
209            List<Presubscription> newPresubscriptions = new LinkedList<>();
210            for (Resource resourceToSubscribeTo : observableResources) {
211                Presubscription presubscription = new Presubscription(resourceToSubscribeTo);
212                newPresubscriptions.add(presubscription);
213                log("New presubscription", presubscription);
214            }
215            api.updatePresubscriptions(newPresubscriptions);
216            // Listing all defined presubscriptions.
217            listPresubscriptions = api.listPresubscriptions();
218            for (Presubscription presubscription : listPresubscriptions) {
219                log("Newly defined presubscription", presubscription);
220            }
221            // Deleting all presubscriptions present on the system.
222            api.deletePresubscriptions();
223        } catch (Exception e) {
224            logError("last API Metadata", api.getLastApiMetadata());
225            fail(e.getMessage());
226        }
227    }

299     * Subscribes to a resource with callbacks.
300     */
301    @SuppressWarnings({ "boxing", "null" })
302    @Example
303    public void subscribeToResourcesWithCallbacks() {
304        ConnectionOptions config = Configuration.get();
305        Connect api = new Connect(config);
306        try {
307            // Getting a connected device.
308            DeviceListOptions options = new DeviceListOptions();
309            options.setLimit(1);
310            Paginator<Device> deviceIterator = api.listAllConnectedDevices(options);
311            if (!deviceIterator.hasNext()) {
312                fail("No endpoints registered. Aborting.");
313            }
314            Device device = deviceIterator.next();
315            log("Device", device);
316            List<Resource> observableResources = api.listObservableResources(device);
317            if (observableResources == null) {
318                fail("There is no observable resources on this device");
319            }
320            for (final Resource resourceToSubscribeTo : observableResources) {
321                log("Resource suscribed", resourceToSubscribeTo);
322                // Creating a subscriber for each resource.
323                if (resourceToSubscribeTo != null) {
324                    // Defining callbacks.
325                    Callback<Object> onNotificationCallback = new Callback<Object>() {
326
327                        @Override
328                        public void execute(Object arg) {
329                            log("Received notification value for " + resourceToSubscribeTo + " using callbacks", arg);
330
331                        }
332                    };
333                    Callback<Throwable> onErrorCallback = new Callback<Throwable>() {
334
335                        @Override
336                        public void execute(Throwable t) {
337                            log("Received following error for " + resourceToSubscribeTo, t);
338
339                        }
340                    };
341                    api.addResourceSubscription(resourceToSubscribeTo, onNotificationCallback, onErrorCallback);
342                }
343            }
344            // Listening to notifications for 2 minutes.
345            api.startNotifications();
346            Thread.sleep(120000);
347            // Stopping notification pull channel.
348            api.stopNotifications();
349            Thread.sleep(100);
350            api.shutdownConnectService();
351
352        } catch (Exception e) {
353            e.printStackTrace();
354            logError("last API Metadata", api.getLastApiMetadata());
355            try {
356                api.stopNotifications();
357                Thread.sleep(100);
358            } catch (Exception e1) {
359                e1.printStackTrace();
360            }
361            api.shutdownConnectService();
362            fail(e.getMessage());
363        }
364    }

422     * <p>
423     * Note: This example introduces new high level abstraction objects such as Observer
424     */
425
426    @Example
427    public void subscribeToDeviceStateChanges() {
428        ConnectionOptions config = Configuration.get();
429        Connect api = new Connect(config);
430        try {
431            // Creating an Observer listening to device state changes for devices whose ids start with 016 and for
432            // devices which are newly registered or expired.
433            // For more information about backpressure strategies, please have a look at related documentation:
434            // https://github.com/ReactiveX/RxJava/wiki/Backpressure
435            DeviceStateObserver observer = api.subscribe().deviceState(
436                    DeviceStateFilterOptions.newFilter().likeDevice("016%")
437                            .inDeviceStates(Arrays.asList(DeviceState.REGISTRATION, DeviceState.EXPIRED_REGISTRATION)),
438                    BackpressureStrategy.BUFFER);
439            // Printing device changes when they happen.
440            observer.flow().subscribe(System.out::println);
441            // Listening to device state changes for 2 minutes.
442            Thread.sleep(120000);
443
444            // Stopping notification pull channel.
445            api.stopNotifications();
446            Thread.sleep(100);
447            api.shutdownConnectService();
448
449        } catch (Exception e) {
450            e.printStackTrace();
451            logError("last API Metadata", api.getLastApiMetadata());
452            try {
453                api.stopNotifications();
454                Thread.sleep(100);
455            } catch (Exception e1) {
456                e1.printStackTrace();
457            }
458            api.shutdownConnectService();
459            fail(e.getMessage());
460        }

518    /**
519     * Sets up a webhook for notifications
520     */
521    @Example
522    public void setUpAWebhook() {
523        ConnectionOptions config = Configuration.get();
524        Connect api = new Connect(config);
525        // Telling the API to stop notification channel if already in use
526        api.setForceClear(true);
527        try {
528            // Defining resource to listen to
529            String resourcePath = "/5002/0/1";
530            log("Resource path of interest", resourcePath);
531            // Creating webhook.
532            Webhook webhook = new Webhook(new URL("http://mbedcloudjavawebhooktest.requestcatcher.com/test"));
533            log("Webhook", webhook);
534            // Setting up webhook.
535            api.updateWebhook(webhook);
536            Thread.sleep(2000);
537            // Adding subscription to all connected devices.
538            Paginator<Device> connectedDevices = api.listAllConnectedDevices(null);
539            for (Device connectedDevice : connectedDevices) {
540                try {
541                    Resource resource = api.getResource(connectedDevice, resourcePath);
542                    if (resource != null) {
543                        api.addResourceSubscription(resource);
544                    }
545                } catch (Exception e1) {
546                    e1.printStackTrace();
547                    logError("An error occurred when trying to fetch Resource [" + resourcePath + "] on device: "
548                            + connectedDevice, api.getLastApiMetadata());
549                }
550            }
551            // Waiting for notifications to be sent to the webhook.
552            Thread.sleep(60000);
553            // Deleting the webhook.
554            deleteWebhook(api);
555        } catch (Exception e) {
556            e.printStackTrace();
557            logError("last API Metadata", api.getLastApiMetadata());
558            deleteWebhook(api);
559            fail(e.getMessage());
560        }
561    }

568        ConnectionOptions config = Configuration.get();
569        Connect api = new Connect(config);
570        try {
571            // Using Notification pull channel.
572            api.startNotifications();
573            Thread.sleep(5000);
574            // Stopping notification pull channel.
575            api.stopNotifications();
576            Thread.sleep(100);
577            // Starting a webhook channel.
578            // Creating a webhook.
579            Webhook webhook = new Webhook(new URL("http://mbedcloudjavawebhooktest.requestcatcher.com/test"));
580            log("Webhook", webhook);
581            // Setting up the webhook.
582            api.updateWebhook(webhook);
583            Thread.sleep(2000);
584            // Stopping webhook channel.
585            deleteWebhook(api);
586            // Shutting down connect service.
587            api.shutdownConnectService();
588            Thread.sleep(100);
589        } catch (Exception e) {
590            e.printStackTrace();
591            try {
592                api.stopNotifications();
593                Thread.sleep(100);
594                api.shutdownConnectService();
595                Thread.sleep(100);
596            } catch (Exception e1) {
597                e1.printStackTrace();
598            }
599            deleteWebhook(api);
600        }
