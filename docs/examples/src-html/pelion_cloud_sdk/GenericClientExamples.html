021    /**
022     * Uses the generic client to list users.
023     */
024
025    @Example
026    public void useTheGenericClient() {
027        // example: custom api call
028        // Create a generic client
029        GenericClient client = GenericClient.newClient(Configuration.get());
030        // Define how to generate the request based on a set of parameters
031        CloudRequest<UserListResponse, PelionApi> requestDefinition = new CloudRequest<UserListResponse, PelionApi>() {
032
033            @Override
034            public com.arm.mbed.cloud.sdk.common.CloudRequest.CloudCall<UserListResponse>
035                   defineCall(PelionApi service, Object... parameters) throws MbedCloudException {
036
037                return new CloudCall<GenericClientExamples.UserListResponse>() {
038
039                    @Override
040                    public Call<UserListResponse> call() {
041                        return service.listAllUsers((Integer) parameters[0]);
042                    }
043
044                };
045            }
046
047        };
048        // cloak
049        try {
050
051            @SuppressWarnings("boxing")
052            // uncloak
053            // Make the call with the following set of parameters. here, limit = 2.
054            UserListResponse response = client.callApi(requestDefinition, 2);
055            for (User user : response.getData()) {
056                // Do something with users
057                System.out.println(user);
058            }
059            // cloak
060        } catch (Exception e) {
061            // Logs information about the last API call.
062            logError("last API Metadata", client.getLastApiMetadata());
063            fail(e.getMessage());
064        }
065        // uncloak
066    }
067
068    /**
069     * Definition of the REST endpoint.
070     * <p>
071     * See the documentation from <a href="https://square.github.io/retrofit/">Retrofit</a> to see how to defined a
072     * service.
073     */
074    private interface PelionApi {
075        @GET("v3/users")
076        Call<UserListResponse> listAllUsers(@retrofit2.http.Query("limit") Integer limit);
077    }
078
079    /**
080     * Definition of the response object.
081     * <p>
082     * Note: the response object has to be an entity and hence, should implement {@link SdkModel}.
083     *
084     */
085    private static class UserListResponse implements SdkModel {
086        /**
087         *
088         */
089        private static final long serialVersionUID = 1L;
090        List<User> data;
091
092        /**
093         * @param data
094         */
095        public UserListResponse(List<User> data) {
096            super();
097            this.data = data;
098        }
099
100        public List<User> getData() {
101            return data;
102        }
103
104        public void setData(List<User> data) {
105            this.data = data;
106        }
107
108        @Override
109        public UserListResponse clone() {
110            return new UserListResponse(data);
111        }
112
113        @Override
114        public boolean isValid() {
115            return true;
116        }
117
118        @Override
119        public String getId() {
120            return null;
121        }
