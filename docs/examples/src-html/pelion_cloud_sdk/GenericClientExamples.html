028    // an example: custom api call
029    /**
030     * Definition of the REST endpoint.
031     * <p>
032     * See the documentation from <a href="https://square.github.io/retrofit/">Retrofit</a> to see how to defined a
033     * service.
034     */
035    private interface PelionApi {
036        @GET("v3/users")
037        Call<UserListResponse> listAllUsers(@retrofit2.http.Query("limit") Integer limit);
038    }
039
040    /**
041     * Definition of the response object.
042     */
043    private static class UserListResponse {
044        List<User> data;
045
046        public List<User> getData() {
047            return data;
048        }
049
050        @SuppressWarnings("unused")
051        public void setData(List<User> data) {
052            this.data = data;
053        }
054
055    }
056
057    // cloak
058    @Example
059    public void useTheGenericClient() {
060        // uncloak
061        // Create a generic client
062        GenericClient client = GenericClient.newClient(Configuration.get());
063        // Define how to generate the request based on a set of parameters
064        CloudRequest<UserListResponse, PelionApi> requestDefinition = new CloudRequest<UserListResponse, PelionApi>() {
065
066            @Override
067            public com.arm.mbed.cloud.sdk.common.CloudRequest.CloudCall<UserListResponse>
068                   defineCall(PelionApi service, Object... parameters) throws MbedCloudException {
069
070                return new CloudCall<GenericClientExamples.UserListResponse>() {
071
072                    @Override
073                    public Call<UserListResponse> call() {
074                        return service.listAllUsers((Integer) parameters[0]);
075                    }
076
077                };
078            }
079
080        };
081        // cloak
082        try {
083            // uncloak
084            // Make the call with the following set of parameters. here, limit = 2.
085            UserListResponse response = client.callApi(requestDefinition, Integer.valueOf(2));
086            for (User user : response.getData()) {
087                // Do something with users
088                System.out.println(user);
089            }
090            // cloak
091        } catch (Exception e) {
092            // Logs information about the last API call.
093            logError("last API Metadata", client.getLastApiMetadata());
094            fail(e.getMessage());
095        }
096    }
097    // uncloak
098
099    // end of example
100    /**
101     * Uses the generic client to list users and pagination.
102     */
103
104    // an example: custom paginated api call
105    /**
106     * Definition of the REST endpoint.
107     * <p>
108     * See the documentation from <a href="https://square.github.io/retrofit/">Retrofit</a> to see how to defined a
109     * service.
110     * <p>
111     * Note: the response object has to be an page ie {@link ListResponse}.
112     */
113    private interface PelionListApi {
114        @GET("v3/users")
115        Call<ListResponse<User>> listAllUsers(@retrofit2.http.Query("limit") Integer limit);
116    }
117
118    // cloak
119    @Example
120    public void useGenericClientWithPagination() {
121        // uncloak
122        // Create a generic client
123        try (Sdk sdk = Sdk.createSdk(Configuration.get())) {
124
125            // Define how to generate the request based on a set of parameters
126
127            CloudRequest.CloudListRequest<User, PelionListApi,
128                                          ListOptions> requestDefinition = new CloudRequest.CloudListRequest<User,
129                                                                                                             PelionListApi,
130                                                                                                             ListOptions>() {
131
132                                              @Override
133                                              public CloudCall<ListResponse<User>>
134                                                     defineCall(PelionListApi service, ListOptions options,
135                                                                Object... extraParameters) throws MbedCloudException {
136                                                  return new CloudCall<ListResponse<User>>() {
137
138                                                      @Override
139                                                      public Call<ListResponse<User>> call() {
140                                                          return service.listAllUsers(options.getPageSize());
141                                                      }
142                                                  };
143                                              }
144
145                                          };
146            // Make the call with the following set of parameters. here, pageSize = 2.
147            Paginator<User> paginatedResponse = sdk.genericClient().callPaginatedApi(requestDefinition,
148                                                                                     new ListOptions().pageSize(2));
149            // Do something with users
150            paginatedResponse.forEach(System.out::println);
151
152        }
153        // cloak
154        catch (Exception e) {
155            fail(e.getMessage());
156        }
157        // uncloak
158    }
159
160    // end of example
