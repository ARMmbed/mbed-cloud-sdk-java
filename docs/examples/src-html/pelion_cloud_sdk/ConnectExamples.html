185    /**
186     * Creates, updates and deletes a presubscription.
187     */
188    @SuppressWarnings("null")
189    @Example
190    public void managePresubscription() {
191        ConnectionOptions config = Configuration.get();
192        Connect api = new Connect(config);
193        try {
194            // Listing already defined presubscriptions.
195            List<Presubscription> listPresubscriptions = api.listPresubscriptions();
196            for (Presubscription presubscription : listPresubscriptions) {
197                log("Already defined presubscription", presubscription);
198            }
199            // Getting a connected device.
200            DeviceListOptions options = new DeviceListOptions();
201            options.setLimit(Integer.valueOf(1));
202            Paginator<Device> deviceIterator = api.listAllConnectedDevices(options);
203            if (!deviceIterator.hasNext()) {
204                fail("No endpoints registered. Aborting.");
205            }
206            Device device = deviceIterator.next();
207            log("Device", device);
208            // Adding presubscriptions on all observable resources of a device.
209            List<Resource> observableResources = api.listObservableResources(device);
210            if (observableResources == null) {
211                fail("There is no observable resources on this device");
212            }
213            List<Presubscription> newPresubscriptions = new LinkedList<>();
214            for (Resource resourceToSubscribeTo : observableResources) {
215                Presubscription presubscription = new Presubscription(resourceToSubscribeTo);
216                newPresubscriptions.add(presubscription);
217                log("New presubscription", presubscription);
218            }
219            api.updatePresubscriptions(newPresubscriptions);
220            // Listing all defined presubscriptions.
221            listPresubscriptions = api.listPresubscriptions();
222            for (Presubscription presubscription : listPresubscriptions) {
223                log("Newly defined presubscription", presubscription);
224            }
225            // Deleting all presubscriptions present on the system.
226            api.deletePresubscriptions();
227        } catch (Exception e) {
228            logError("last API Metadata", api.getLastApiMetadata());
229            fail(e.getMessage());
230        }
231    }

676            api.startNotifications();
677            Thread.sleep(5000);
678            // Shutting down connect service.
679            api.shutdownConnectService();
680            Thread.sleep(100);
681        } catch (Exception e) {
682            e.printStackTrace();
683            try {
684                api.shutdownConnectService();
685                Thread.sleep(100);
686            } catch (Exception e1) {
687                e1.printStackTrace();
688            }
689            deleteWebhook(api);
690        }
691    }

695            api.deleteWebhook();
696        } catch (MbedCloudException e) {
697            fail(e.getMessage());
698        }
