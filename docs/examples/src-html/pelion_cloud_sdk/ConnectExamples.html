221    /**
222     * Subscribes to device state changes.
223     * <p>
224     * Note: This example introduces new high level abstraction objects such as Observer
225     */
226
227    @Example
228    public void subscribeToDeviceStateChanges() {
229        try (Sdk sdk = Sdk.createSdk(Configuration.get())) {
230            // Creating an Observer listening to device state changes for devices whose ids start with 016 and for
231            // devices which are newly registered or expired.
232            // For more information about backpressure strategies, please have a look at related documentation:
233            // https://github.com/ReactiveX/RxJava/wiki/Backpressure
234            DeviceStateObserver observer = sdk.subscribe(DeviceStateFilterOptions.newFilter().likeDevice("016%")
235                                                                                 .inDeviceStates(Arrays.asList(com.arm.mbed.cloud.sdk.subscribe.model.DeviceState.REGISTRATION,
236                                                                                                               com.arm.mbed.cloud.sdk.subscribe.model.DeviceState.EXPIRED_REGISTRATION)),
237                                                         BackpressureStrategy.BUFFER);
238            // Printing device changes when they happen.
239            observer.flow().subscribe(System.out::println);
240            // Listening to device state changes for 2 minutes.
241            Thread.sleep(120000);
242        } catch (Exception e) {
243            e.printStackTrace();
244            fail(e.getMessage());
245        }
246    }

248    /**
249     * Subscribes to resource value changes.
250     * <p>
251     * Note: This example introduces new high level abstraction objects such as Observer
252     */
253
254    @Example
255    public void subscribeToResourceValueChanges() {
256        try (Sdk sdk = Sdk.createSdk(Configuration.get())) {
257            // Creating an Observer listening to resource value changes for devices whose ids start with 016 and
258            // resource paths starting with /3/0/.
259            // For more information about backpressure strategies, please have a look at related documentation:
260            // https://github.com/ReactiveX/RxJava/wiki/Backpressure
261            // For more information about First Value strategies, have a look at
262            // com.arm.mbed.cloud.sdk.subscribe.model.FirstValue
263            ResourceValueObserver observer = sdk.subscribe(SubscriptionFilterOptions.newFilter().likeDevice("016%")
264                                                                                    .likeResourcePath("/3/0/%"),
265                                                           BackpressureStrategy.BUFFER, FirstValue.ON_VALUE_UPDATE);
266            // Printing resource value notification when they happen.
267            observer.flow().subscribe(System.out::println);
268            // Listening to resource value changes for 2 minutes.
269            Thread.sleep(120000);
270
271        } catch (Exception e) {
272            e.printStackTrace();
273            fail(e.getMessage());
274        }

288            String resourcePath = "/3200/0/5501";
289            NotificationMessage notifications = new NotificationMessage();
290            for (String payload : payloadList) {
291                NotificationData notification = new NotificationData();
292                notification.setEp(deviceId);
293                notification.setPath(resourcePath);
294                notification.setPayload(payload);
295                notifications.addNotificationsItem(notification);
296            }
297            // Creating the same notifications but using their JSON representation instead.
298            String otherNotifications = "{\"notifications\":[{\"path\":\"/3200/0/5501\",\"payload\":\"Q2hhbmdlIG1lIQ\u003d\u003d\",\"ep\":\"015f4ac587f500000000000100100249\"},{\"path\":\"/3200/0/5501\",\"payload\":\"VGhpcyBpcyB2YWx1ZSAy\",\"ep\":\"015f4ac587f500000000000100100249\"}"

321    @Example
322    public void useServerInitiatedNotificationChannel() {
323        // an example: using a webhook for handling notifications from Pelion Cloud
324        // Ensure no daemon threads are started.
325        try (Sdk sdk = Sdk.createSdk(Configuration.get().autostartDaemon(false))) {
326            // Describing a webhook to use (i.e. a publicly available webserver accepting POST request).
327            Webhook webhook = new Webhook(new URL("http://mbedcloudjavawebhooktest.requestcatcher.com/test"));
328            log("Webhook", webhook);
329            // Registering the webhook to Pelion Cloud.
330            sdk.lowLevelRest().getConnectModule().updateWebhook(webhook);
331            // Defining a resource to listen to
332            String resourcePath = "/5002/0/1";
333            log("Resource path of interest", resourcePath);
334            // Subscribing to the resource value changes
335            sdk.subscribe(SubscriptionFilterOptions.newFilter().equalResourcePath(resourcePath),
336                          BackpressureStrategy.MISSING)
337               .flow().subscribe();
338            // Ensuring the webhook has been correctly registered
339            log("Registered webhook", sdk.lowLevelRest().getConnectModule().getWebhook());
340            // Waiting for notifications to be sent to the webhook.
341            Thread.sleep(60000);// TODO do some actual work in your application
342            // Unlink the webhook.
343            sdk.lowLevelRest().getConnectModule().deleteWebhook();
