161    @Example
162    public void manageSubtenants() {
163        // an example: creating and managing a subtenant account
164        try (Sdk sdk = Sdk.createSdk(Configuration.get()); AccountDao myAccountDao = sdk.foundation().getAccountDao();
165             SubtenantUserDao userDao = sdk.foundation().getSubtenantUserDao();) {
166            // Fetch my account
167            myAccountDao.me();
168            // Create a new User
169            SubtenantUser user = new SubtenantUser();
170            user.setAccountId(myAccountDao.getId());
171            user.setFullName("tommi the wombat");
172            user.setEmail("tommi_the_wombat@email.com");
173            user.setUsername("tommi_wombat");
174            user.setPhoneNumber("0800001066");
175            // cloak
176            // Perform some account cleanup
177            performAccountCleanup(myAccountDao, user, userDao);
178            // uncloak
179            userDao.create(user);
180            // Find this added user in my account user list.
181            String createdId = userDao.getId();
182            // cloak
183            assertNotNull(createdId);
184            // uncloak
185            // Fetch latest details of the current sub-tenant user.
186            userDao.read();
187            // Fetching a user from the list of account users
188            SubtenantUser correspondingUser = StreamSupport.stream(myAccountDao.allUsers(null).spliterator(), false)
189                                                           .filter(u -> u.getId().equals(createdId)).findFirst().get();
190            // cloak
191            assertEquals(userDao.getModel(), correspondingUser);
192            // uncloak
193            // Delete the created user.
194            userDao.delete();
195
196        } catch (MbedCloudException | IOException exception) {
197            // TODO do something with the exception
198            exception.printStackTrace();
199            // cloak
200            fail(exception.getMessage());
201            // uncloak
202        }
203        // end of example
204    }
205
206    private void performAccountCleanup(AccountDao myAccountDao, SubtenantUser user,
207                                       SubtenantUserDao userDao) throws MbedCloudException {
208        StreamSupport.stream(myAccountDao.allUsers(null).spliterator(), false)
209                     .filter(u -> user.getUsername().equals(u.getUsername())).forEach(u -> {
210                         try {
211                             userDao.delete(u);
212                         } catch (MbedCloudException exception) {
213                             exception.printStackTrace();
214                         }
215                     });
216    }

218    /**
219     * Renews a device certificate
220     */
221    @Example
222    public void renewCertificate() {
223        // an example: certificate renew
224        try (Sdk sdk = Sdk.createSdk(Configuration.get()); DeviceDao deviceDao = sdk.foundation().getDeviceDao()) {
225            // Find the certificate issuers configuration in use. In this case, it is known that the reference is
226            // "LWM2M".
227            CertificateIssuerConfig myConfig = StreamSupport.stream(sdk.foundation().getCertificateIssuerConfigListDao()
228                                                                       .getPaginator().spliterator(),
229                                                                    false)
230                                                            .filter(c -> c.getReference().equals("LWM2M")).findFirst()
231                                                            .get();
232            // Renew the certificate of all connected devices
233            StreamSupport.stream(sdk.foundation().getDeviceListDao().paginator().spliterator(), false)
234                         .filter(d -> d.getState() == DeviceState.REGISTERED).forEach(d -> {
235                             // Configure the DAO with the model of interest
236                             try {
237                                 deviceDao.setModel(d);
238                                 deviceDao.renewCertificate(myConfig.getReference());
239                                 log(myConfig + " was renewed on " + d);
240                             } catch (@SuppressWarnings("unused") MbedCloudException exception) {
241                                 // TODO do something with the exception
242                                 log(myConfig + " could not be renewed on " + d);
243                             }
244
245                         });
246
247        } catch (MbedCloudException | IOException exception) {
248            // TODO do something with the exception
249            exception.printStackTrace();
250            // cloak
251            fail(exception.getMessage());
252            // uncloak
253        }
254        // end of example
255    }

261    public void enrolDevicesInBulk() {
262        ClassLoader classLoader = getClass().getClassLoader();
263        File file = null;
264        try {
265            file = File.createTempFile("test", ".csv");
266            file.deleteOnExit();
267            Files.copy(classLoader.getResource("test.csv").openStream(), file.toPath(),
268                       StandardCopyOption.REPLACE_EXISTING);
269        } catch (IOException exception1) {
270            // TODO Auto-generated catch block
271            exception1.printStackTrace();
272        }
273        assertNotNull(file);
274        assertTrue(file.exists());
275        // an example: device enrollment bulk
276        try (Sdk sdk = Sdk.createSdk(Configuration.get());
277             DeviceEnrollmentBulkCreateDao enrolmentDao = sdk.foundation().getDeviceEnrollmentBulkCreateDao()) {
278            // file is a csv file containing all the devices which need to be enrolled. See test.csv in the /resources
279            // folder as example.
280            enrolmentDao.create(new DataFile(file));
281            // cloak
282            assertEquals(DeviceEnrollmentBulkCreateStatus.NEW, enrolmentDao.getModel().getStatus());
283            // uncloak
284            // read current state of bulk enrollment
285            enrolmentDao.read();
286            // cloak
287            assertTrue(enrolmentDao.getModel().getStatus() == DeviceEnrollmentBulkCreateStatus.PROCESSING
288                       || enrolmentDao.getModel().getStatus() == DeviceEnrollmentBulkCreateStatus.COMPLETED);
289            // uncloak
290            // Wait until the report is ready
291            // FIXME this is just a cruddy example of waiting for the bulk operation to complete. Better handling should
292            // be performed in production application.
293            while (enrolmentDao.getModel().getStatus() != DeviceEnrollmentBulkCreateStatus.COMPLETED) {
294                Thread.sleep(1000);
295                enrolmentDao.read();
296            }
297            // Download the enrolment report files
298            FileDownload report = enrolmentDao.downloadFullReportFile((String) null);
299            log("report", report);
300            FileDownload errorReport = enrolmentDao.downloadErrorsReportFile((String) null);
301            log("errorReport", errorReport);
302
303        } catch (MbedCloudException | InterruptedException exception) {
304            // TODO do something with the exception
305            exception.printStackTrace();
306            // cloak
307            fail(exception.getMessage());
308            // uncloak
309        }
310        // end of example
311        catch (IOException exception1) {
312            // TODO Auto-generated catch block
313            exception1.printStackTrace();
314        }
315    }

317    /**
318     * Renews a device certificate
319     */
320    @Example
321    public void blacklistCertificates() {
322        try (Sdk sdk = Sdk.createSdk(Configuration.get());
323             TrustedCertificateListDao certificateDao = sdk.foundation().getTrustedCertificateListDao()) {
324            // Find a production certificate
325            TrustedCertificate certificateOfInterest = certificateDao.list(new TrustedCertificateListOptions().notEqualToDeviceExecutionMode(1))
326                                                                     .first();
327            assertNotNull(certificateOfInterest);
328            String myCertId = certificateOfInterest.getId();
329            // Record the original status to revert to its original state at the end
330            TrustedCertificateStatus originalStatus = certificateDao.getNewModelDao().read(myCertId).getStatus();
331            // an example: certificate black listing
332
333            // Set the certificate to inactive
334            TrustedCertificateDao myCertificateDao = sdk.foundation().getTrustedCertificateDao();
335            myCertificateDao.read(myCertId);
336            myCertificateDao.getModel().setStatus(TrustedCertificateStatus.INACTIVE);
337            myCertificateDao.update();
338
339            // List all devices which have tried to bootstrap
340            sdk.foundation().getDeviceEnrollmentDenialListDao()
341               .list(new DeviceEnrollmentDenialListOptions().equalToTrustedCertificateId(myCertId))
342               .forEach(d -> log("Device endpoint name: " + d.getEndpointName()));
343
344            // end of example
345            myCertificateDao.read();
346            assertEquals(TrustedCertificateStatus.INACTIVE, myCertificateDao.getModel().getStatus());
347            // Revert the certificate to its original status
348            myCertificateDao.getModel().setStatus(originalStatus);
349            myCertificateDao.update();
350            myCertificateDao.read();
351            assertEquals(originalStatus, myCertificateDao.getModel().getStatus());
352            myCertificateDao.close();
353        } catch (MbedCloudException | IOException exception) {
354            // TODO do something with the exception
355            exception.printStackTrace();
356
357            fail(exception.getMessage());
358
359        }
360    }

362    /**
363     * Creates and starts a firmware update campaign
364     */
365    @Example
366    public void launchUpdateCampaign() {
367        try (Sdk sdk = Sdk.createSdk(Configuration.get());
368             FirmwareManifestListDao manifestDao = sdk.foundation().getFirmwareManifestListDao()) {
369            // Find a production manifest
370            FirmwareManifest manifestOfInterest = manifestDao.list(new FirmwareManifestListOptions().maxResults(2))
371                                                             .first();
372            assertNotNull(manifestOfInterest);
373            String myManifestId = manifestOfInterest.getId();
374            // an example: firmware update campaign launch
375            UpdateCampaignDao myCampaignDao = sdk.foundation().getUpdateCampaignDao();
376            // Define an update campaign
377            UpdateCampaign newCampaign = new UpdateCampaign();
378            newCampaign.setName("campaign - " + (new Date().toString()));
379            newCampaign.setDescription("Update campaign for prior 2019 devices");
380            newCampaign.setRootManifestId(myManifestId);
381            // All devices created before the 1st of January 2019 will have their firmware replaced by the one defined
382            // in the manifest
383            newCampaign.setDeviceFiltersHelper(new DeviceListOptions().lessThanCreatedAt(new GregorianCalendar(2019, 0,
384                                                                                                               1).getTime())
385                                                                      .getFilter());
386            // Create the campaign
387            myCampaignDao.create(newCampaign);
388            // Determine the phase of the campaign
389            System.out.println(myCampaignDao.read().getPhase());
390            // Start the campaign
391            myCampaignDao.start();
392            // Determine the phase of the campaign
393            System.out.println(myCampaignDao.read().getPhase());
394            // Print all device metadata related to this campaign
395            myCampaignDao.allDeviceMetadata(null).all().forEach(System.out::println);
396            // end of example
397            myCampaignDao.stop();
398            myCampaignDao.delete();
399            myCampaignDao.close();
400        } catch (MbedCloudException | IOException exception) {
401            // TODO do something with the exception
402            exception.printStackTrace();
403
404            fail(exception.getMessage());
405
406        }
